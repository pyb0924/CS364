OUTPUT MACRO INFO,DATA,NUM;USE: DX(info),DI(data),SI(numï¼‰
	LOCAL RES_OUTPUT
	STRING_OUT  INFO;output hint info
	MOV SI,NUM
	LEA DI,DATA	
	RES_OUTPUT:
	DATA_OUTPUT DI
	CHAR_OUT 20H
	ADD DI,4
	DEC SI
	CMP SI,0
	JNZ RES_OUTPUT
ENDM
DATA_OUTPUT MACRO YOURDATA
	LOCAL NUM_TO_STRING_PUSH
	LOCAL NUM_TO_STRING_POP
		MOVZX AX,[YOURDATA]
		MOV SIGN,0
		TEST AL,80H
		MOV CX,0
		JZ NUM_TO_STRING_PUSH
		; process negative num
		MOV SIGN,2DH
		NEG AL	;change to unsigned num	
	NUM_TO_STRING_PUSH:
		MOV DL,10
		DIV DL; !
		ADD AH,30H
		MOVZX DX,AH
		PUSH DX
		MOV AH,0
		INC CX; n-bit decimal
		CMP AX,0
		JNZ NUM_TO_STRING_PUSH
		MOV DL,SIGN
		CMP DL,0
		JZ NUM_TO_STRING_POP
		CHAR_OUT DL
	NUM_TO_STRING_POP:
		POP DX
		CHAR_OUT DL
		LOOP NUM_TO_STRING_POP
ENDM
